問題2: コミック — 「プライベート...」フォルダ選択不可 & ZIP クラッシュ
原因（プライベート表示）: SAF の ACTION_OPEN_DOCUMENT_TREE が返す URI に対し、一部のデバイスで内部ストレージのプライベート領域を指すツリーが表示される。これは Android の仕様で完全に排除できないが、ユーザーが選択した後のバリデーションが不十分。

原因（ZIP クラッシュ）: loadArchivePages() 内で extractZipImages() → buildPageModelsFromFiles() → processAndMaterializePages() が呼ばれるが、この一連の処理がメインスレッド制約なしに大量のビットマップをデコードし、OutOfMemoryError またはバッファ不足で未捕捉例外が発生する。また、ZipInputStream のエントリ名に日本語パスが含まれると sanitizeFileName だけでは不十分で、パス区切りの処理が壊れることがある。

C. processAndMaterializePages() にサブサンプリングを追加してメモリ使用量削減

Copyprivate suspend fun processAndMaterializePages(
    sourceName: String,
    openStream: () -> java.io.InputStream?,
    settings: ComicReaderSettings
): List<Uri> {
    val sourceBytes = withContext(Dispatchers.IO) {
        openStream()?.use { it.readBytes() }
    } ?: return emptyList()

    val processed = try {
        pageProcessor.process(
            inputStream = ByteArrayInputStream(sourceBytes),
            settings = settings
        )
    } catch (e: Exception) {
        android.util.Log.w("ComicReader", "PageProcessor failed for $sourceName", e)
        emptyList()
    }

    if (processed.isEmpty()) {
        return writeSingleTempPage(sourceBytes, sourceName)
    }

    val bounds = BitmapFactory.Options().apply { inJustDecodeBounds = true }
    BitmapFactory.decodeByteArray(sourceBytes, 0, sourceBytes.size, bounds)
    val fullRect = android.graphics.Rect(0, 0, bounds.outWidth, bounds.outHeight)

    if (processed.size == 1 && processed.first().rect == fullRect) {
        return writeSingleTempPage(sourceBytes, sourceName)
    }

    // サブサンプリング: 4000px超なら縮小
    val sampleSize = if (bounds.outWidth > 4000 || bounds.outHeight > 4000) 2 else 1
    val decodeOptions = BitmapFactory.Options().apply { inSampleSize = sampleSize }
    val sourceBitmap = BitmapFactory.decodeByteArray(sourceBytes, 0, sourceBytes.size, decodeOptions)
        ?: return emptyList()

    val scaleX = sourceBitmap.width.toFloat() / bounds.outWidth
    val scaleY = sourceBitmap.height.toFloat() / bounds.outHeight

    val outputUris = mutableListOf<Uri>()
    processed.forEachIndexed { index, page ->
        val rect = page.rect
        val scaledLeft = (rect.left * scaleX).toInt().coerceAtLeast(0)
        val scaledTop = (rect.top * scaleY).toInt().coerceAtLeast(0)
        val scaledWidth = (rect.width() * scaleX).toInt().coerceIn(1, sourceBitmap.width - scaledLeft)
        val scaledHeight = (rect.height() * scaleY).toInt().coerceIn(1, sourceBitmap.height - scaledTop)

        val cropped = Bitmap.createBitmap(sourceBitmap, scaledLeft, scaledTop, scaledWidth, scaledHeight)
        val file = File(
            ensureProcessedCacheDir(),
            "${sourceName.hashCode()}_${System.nanoTime()}_${index}.jpg"
        )
        FileOutputStream(file).use { output ->
            cropped.compress(Bitmap.CompressFormat.JPEG, 90, output)
        }
        cropped.recycle()
        outputUris += file.toUri()
    }
    sourceBitmap.recycle()
    return outputUris
}

根本原因: NetworkDownloadManager はキューの状態管理のみを行っており、実際にファイルをダウンロードするロジックが存在しない。enqueueDownload() は DownloadTask を PENDING 状態でリストに追加するだけ。DownloadService も通知を更新するだけで、実際のダウンロード処理を開始していない。キュー削除も cancel() でステータスを CANCELED に変更するだけでリストから除去しない。

A. NetworkDownloadManager.kt を全面置換

Copypackage com.example.litemediaplayer.network

import android.content.Context
import android.net.Uri
import android.util.Log
import com.hierynomus.msdtyp.AccessMask
import com.hierynomus.mssmb2.SMB2CreateDisposition
import com.hierynomus.mssmb2.SMB2ShareAccess
import com.hierynomus.smbj.SMBClient
import com.hierynomus.smbj.auth.AuthenticationContext
import com.hierynomus.smbj.share.DiskShare
import kotlinx.coroutines.*
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.asStateFlow
import kotlinx.coroutines.flow.update
import okhttp3.OkHttpClient
import okhttp3.Request
import java.io.InputStream
import java.util.UUID
import java.util.concurrent.ConcurrentHashMap

class NetworkDownloadManager {
    data class DownloadTask(
        val id: String = UUID.randomUUID().toString(),
        val sourceUri: String,
        val server: NetworkServer,
        val destinationUri: Uri,
        val fileName: String,
        val totalBytes: Long,
        val downloadedBytes: Long = 0L,
        val status: DownloadStatus = DownloadStatus.PENDING
    )

    enum class DownloadStatus {
        PENDING, DOWNLOADING, PAUSED, COMPLETED, FAILED, CANCELED
    }

    private val _downloads = MutableStateFlow<List<DownloadTask>>(emptyList())
    val downloads: StateFlow<List<DownloadTask>> = _downloads.asStateFlow()

    private val scope = CoroutineScope(Dispatchers.IO + SupervisorJob())
    private val activeJobs = ConcurrentHashMap<String, Job>()
    private val httpClient = OkHttpClient.Builder().build()
    private var appContext: Context? = null
    private var maxConcurrent: Int = 2

    fun init(context: Context) {
        appContext = context.applicationContext
    }

    fun setMaxConcurrent(max: Int) {
        maxConcurrent = max.coerceIn(1, 5)
        processQueue()
    }

    fun enqueueDownload(
        sourceUri: String,
        server: NetworkServer,
        destinationUri: Uri,
        fileName: String,
        totalBytes: Long
    ): String {
        val task = DownloadTask(
            sourceUri = sourceUri,
            server = server,
            destinationUri = destinationUri,
            fileName = fileName,
            totalBytes = totalBytes
        )
        _downloads.update { listOf(task) + it }
        processQueue()
        return task.id
    }

    fun pause(taskId: String) {
        activeJobs[taskId]?.cancel()
        activeJobs.remove(taskId)
        updateTask(taskId) { it.copy(status = DownloadStatus.PAUSED) }
    }

    fun resume(taskId: String) {
        updateTask(taskId) { it.copy(status = DownloadStatus.PENDING) }
        processQueue()
    }

    fun cancel(taskId: String) {
        activeJobs[taskId]?.cancel()
        activeJobs.remove(taskId)
        _downloads.update { tasks -> tasks.filter { it.id != taskId } }
    }

    fun removeCompleted() {
        _downloads.update { tasks ->
            tasks.filter { it.status != DownloadStatus.COMPLETED && it.status != DownloadStatus.CANCELED && it.status != DownloadStatus.FAILED }
        }
    }

    private fun processQueue() {
        val currentDownloading = _downloads.value.count { it.status == DownloadStatus.DOWNLOADING }
        if (currentDownloading >= maxConcurrent) return

        val slotsAvailable = maxConcurrent - currentDownloading
        val pendingTasks = _downloads.value
            .filter { it.status == DownloadStatus.PENDING }
            .take(slotsAvailable)

        pendingTasks.forEach { task -> startDownload(task) }
    }

    private fun startDownload(task: DownloadTask) {
        updateTask(task.id) { it.copy(status = DownloadStatus.DOWNLOADING) }
        val job = scope.launch {
            try {
                val protocol = runCatching { Protocol.valueOf(task.server.protocol) }
                    .getOrDefault(Protocol.SMB)

                when (protocol) {
                    Protocol.SMB -> downloadSmb(task)
                    Protocol.HTTP, Protocol.WEBDAV -> downloadHttp(task)
                }
                updateTask(task.id) { it.copy(status = DownloadStatus.COMPLETED, downloadedBytes = it.totalBytes) }
            } catch (e: CancellationException) {
                // pause/cancel による中断 — 何もしない
            } catch (e: Exception) {
                Log.e("DownloadManager", "Download failed: ${task.fileName}", e)
                updateTask(task.id) { it.copy(status = DownloadStatus.FAILED) }
            } finally {
                activeJobs.remove(task.id)
                processQueue()
            }
        }
        activeJobs[task.id] = job
    }

    private suspend fun downloadSmb(task: DownloadTask) {
        val ctx = appContext ?: throw IllegalStateException("Context not initialized")
        val shareName = task.server.shareName?.takeIf { it.isNotBlank() }
            ?: throw IllegalStateException("SMB share name not set")
        val remotePath = task.sourceUri.removePrefix("/")

        val client = SMBClient()
        val connection = client.connect(task.server.host, task.server.port ?: 445)
        try {
            val session = connection.authenticate(
                AuthenticationContext(
                    task.server.username.orEmpty(),
                    (task.server.password ?: "").toCharArray(),
                    ""
                )
            )
            val share = session.connectShare(shareName) as? DiskShare
                ?: throw IllegalStateException("Not a disk share")

            val smbFile = share.openFile(
                remotePath,
                setOf(AccessMask.GENERIC_READ),
                null,
                SMB2ShareAccess.ALL,
                SMB2CreateDisposition.FILE_OPEN,
                null
            )

            writeToDestination(ctx, task, smbFile.inputStream)
            smbFile.close()
            share.close()
            session.close()
        } finally {
            runCatching { connection.close() }
            runCatching { client.close() }
        }
    }

    private suspend fun downloadHttp(task: DownloadTask) {
        val ctx = appContext ?: throw IllegalStateException("Context not initialized")
        val server = task.server
        val scheme = "http"
        val hostPort = if (server.port != null) "${server.host}:${server.port}" else server.host
        val base = server.basePath?.trim('/')?.takeIf { it.isNotBlank() }
        val remote = task.sourceUri.removePrefix("/")
        val url = if (base.isNullOrBlank()) "$scheme://$hostPort/$remote"
                  else "$scheme://$hostPort/$base/$remote"

        val requestBuilder = Request.Builder().url(url).get()
        val user = server.username?.takeIf { it.isNotBlank() }
        val pass = server.password?.takeIf { it.isNotBlank() }
        if (user != null && pass != null) {
            val token = java.util.Base64.getEncoder()
                .encodeToString("$user:$pass".toByteArray(Charsets.UTF_8))
            requestBuilder.addHeader("Authorization", "Basic $token")
        }

        val response = httpClient.newCall(requestBuilder.build()).execute()
        if (!response.isSuccessful) throw IllegalStateException("HTTP ${response.code}")
        val body = response.body ?: throw IllegalStateException("Empty response body")

        writeToDestination(ctx, task, body.byteStream())
        response.close()
    }

    private suspend fun writeToDestination(ctx: Context, task: DownloadTask, input: InputStream) {
        // ダウンロード先フォルダ (SAF URI) 内に新ファイルを作成
        val destDir = androidx.documentfile.provider.DocumentFile.fromTreeUri(ctx, task.destinationUri)
        val mimeType = when {
            task.fileName.endsWith(".mp4", true) -> "video/mp4"
            task.fileName.endsWith(".mkv", true) -> "video/x-matroska"
            task.fileName.endsWith(".zip", true) -> "application/zip"
            task.fileName.endsWith(".cbz", true) -> "application/zip"
            task.fileName.endsWith(".cbr", true) -> "application/x-rar-compressed"
            else -> "application/octet-stream"
        }

        val destFile = destDir?.createFile(mimeType, task.fileName)
            ?: throw IllegalStateException("Cannot create file in destination")

        ctx.contentResolver.openOutputStream(destFile.uri)?.use { output ->
            val buffer = ByteArray(8192)
            var totalRead = 0L
            var bytesRead: Int

            while (input.read(buffer).also { bytesRead = it } != -1) {
                ensureActive()  // CancellationException をチェック
                output.write(buffer, 0, bytesRead)
                totalRead += bytesRead
                updateTask(task.id) { it.copy(downloadedBytes = totalRead) }
            }
            output.flush()
        } ?: throw IllegalStateException("Cannot open output stream")
    }

    private fun updateTask(taskId: String, transform: (DownloadTask) -> DownloadTask) {
        _downloads.update { tasks ->
            tasks.map { if (it.id == taskId) transform(it) else it }
        }
    }

    companion object {
        val shared: NetworkDownloadManager by lazy { NetworkDownloadManager() }
    }
}
