問題1: 動画再生ができない
現状のコード: Navigation.kt で Base64 エンコード/デコードは正しく実装済み。PlayerPlaybackScreen の LaunchedEffect(videoUri) で Uri.parse(videoUri) → MediaItem.fromUri(target) → exoPlayer.prepare() も正しい。

残る根本原因: ExoPlayer は SAF の content:// URI を直接再生できるが、パーシスタブル権限が PlayerPlaybackScreen のコンテキストに渡っていない。PlayerViewModel が takePersistableUriPermission を取っているのはフォルダの treeUri だけ。個別の動画ファイルの content://com.android.externalstorage.documents/document/... URI に対しては、SAF のツリー権限から派生した読み取り権限を ExoPlayer に伝える必要がある。もう一つの可能性として、ExoPlayer の DefaultDataSource が content:// の SAF ツリー配下ドキュメントを解決できないケースがある。

private fun resolveVideoItems(treeUriString: String): List<VideoItem> {
    // キャッシュがあればそれを返す
    videoItemsCache[treeUriString]?.let { return it }

    android.util.Log.d("PlayerVM", "Scanning folder: $treeUriString")
    val startMs = System.currentTimeMillis()

    val root = DocumentFile.fromTreeUri(context, Uri.parse(treeUriString)) ?: return emptyList()
    val queue = ArrayDeque<DocumentFile>()
    val files = mutableListOf<DocumentFile>()
    queue.add(root)

    while (queue.isNotEmpty()) {
        val current = queue.removeFirst()
        current.listFiles().forEach { document ->
            when {
                document.isDirectory -> queue.add(document)
                document.isFile && isSupportedVideo(document) -> files.add(document)
            }
        }
    }

    val items = files
        .map { document ->
            VideoItem(
                uri = document.uri,
                displayName = document.name ?: "video",
                size = document.length().coerceAtLeast(0L),
                durationMs = 0L,
                resolution = null,
                dateModified = document.lastModified().coerceAtLeast(0L)
            )
        }
        .sortedBy { it.displayName.lowercase() }

    val elapsed = System.currentTimeMillis() - startMs
    android.util.Log.d("PlayerVM", "Folder scan complete: ${items.size} files in ${elapsed}ms")

    // キャッシュに保存
    videoItemsCache[treeUriString] = items
    return items
}

private fun loadMetadataAsync(folders: List<VideoFolderUi>) {

            }

問題3: コミック ZIP 読み込みでアプリが落ちる
現状のコード: extractZipImages() に try-catch は入ったが、processAndMaterializePages() 内の pageProcessor.process() が inputStream.readBytes() でZIP から展開した画像ファイル全体をメモリに読み込む。大きな画像（4000x6000px 等）が多数あると readBytes() と BitmapFactory.decodeByteArray() でメモリが枯渇する。

さらに PageProcessor.process() 内部でも inputStream.readBytes() が呼ばれ、同じ画像データが2回読み込まれている（processAndMaterializePages 内で readBytes() → pageProcessor.process() 内でも readBytes()）。

Copyprivate suspend fun processAndMaterializePages(
    sourceName: String,
    openStream: () -> java.io.InputStream?,
    settings: ComicReaderSettings
): List<Uri> {
    return withContext(Dispatchers.IO) {
        try {
            val sourceBytes = openStream()?.use { it.readBytes() } ?: return@withContext emptyList()
            android.util.Log.d("ComicReader", "Processing page: $sourceName (${sourceBytes.size} bytes)")

            // まず画像として有効かチェック
            val bounds = BitmapFactory.Options().apply { inJustDecodeBounds = true }
            BitmapFactory.decodeByteArray(sourceBytes, 0, sourceBytes.size, bounds)
            if (bounds.outWidth <= 0 || bounds.outHeight <= 0) {
                android.util.Log.w("ComicReader", "Invalid image: $sourceName (${bounds.outWidth}x${bounds.outHeight})")
                return@withContext emptyList()
            }

            android.util.Log.d("ComicReader", "Image size: ${bounds.outWidth}x${bounds.outHeight}")

            // 分割/トリミングが不要ならバイトをそのままファイルに書き出す（Bitmap デコード不要）
            if (!settings.autoTrimEnabled && !settings.autoSplitEnabled) {
                return@withContext writeSingleTempPage(sourceBytes, sourceName)
            }

            val processed = runCatching {
                pageProcessor.process(
                    inputStream = ByteArrayInputStream(sourceBytes),
                    settings = settings
                )
            }.getOrElse { error ->
                android.util.Log.w("ComicReader", "PageProcessor failed for $sourceName", error)
                emptyList()
            }

            if (processed.isEmpty()) {
                return@withContext writeSingleTempPage(sourceBytes, sourceName)
            }

            val fullRect = android.graphics.Rect(0, 0, bounds.outWidth, bounds.outHeight)
            if (processed.size == 1 && processed.first().rect == fullRect) {
                return@withContext writeSingleTempPage(sourceBytes, sourceName)
            }

            // サブサンプリングでメモリ使用量を制限
            val maxDim = maxOf(bounds.outWidth, bounds.outHeight)
            val sampleSize = when {
                maxDim > 8000 -> 4
                maxDim > 4000 -> 2
                else -> 1
            }
            val decodeOptions = BitmapFactory.Options().apply { inSampleSize = sampleSize }
            val sourceBitmap = BitmapFactory.decodeByteArray(sourceBytes, 0, sourceBytes.size, decodeOptions)
                ?: return@withContext emptyList()

            val scaleX = sourceBitmap.width.toFloat() / bounds.outWidth.toFloat().coerceAtLeast(1f)
            val scaleY = sourceBitmap.height.toFloat() / bounds.outHeight.toFloat().coerceAtLeast(1f)
            val outputUris = mutableListOf<Uri>()

            processed.forEachIndexed { index, page ->
                val rect = page.rect
                val left = (rect.left * scaleX).toInt().coerceIn(0, sourceBitmap.width - 1)
                val top = (rect.top * scaleY).toInt().coerceIn(0, sourceBitmap.height - 1)
                val cropWidth = (rect.width() * scaleX).toInt().coerceIn(1, sourceBitmap.width - left)
                val cropHeight = (rect.height() * scaleY).toInt().coerceIn(1, sourceBitmap.height - top)

                val cropped = Bitmap.createBitmap(sourceBitmap, left, top, cropWidth, cropHeight)
                val file = File(
                    ensureProcessedCacheDir(),
                    "${sourceName.hashCode()}_${System.nanoTime()}_${index}.jpg"
                )
                FileOutputStream(file).use { output ->
                    cropped.compress(Bitmap.CompressFormat.JPEG, 85, output)
                }
                cropped.recycle()
                outputUris += file.toUri()
            }
            sourceBitmap.recycle()
            outputUris
        } catch (e: OutOfMemoryError) {
            android.util.Log.e("ComicReader", "OOM processing page: $sourceName", e)
            System.gc()
            emptyList()
        } catch (e: Exception) {
            android.util.Log.e("ComicReader", "Error processing page: $sourceName", e)
            emptyList()
        }
    }
}

Copypackage com.example.litemediaplayer.core

import android.util.Log
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.asStateFlow
import kotlinx.coroutines.flow.update
import java.text.SimpleDateFormat
import java.util.Date
import java.util.Locale

object AppLogger {
    data class LogEntry(
        val timestamp: Long = System.currentTimeMillis(),
        val level: String,
        val tag: String,
        val message: String
    ) {
        val timeString: String
            get() = SimpleDateFormat("HH:mm:ss.SSS", Locale.getDefault()).format(Date(timestamp))

        override fun toString(): String = "[$timeString] $level/$tag: $message"
    }

    private const val MAX_ENTRIES = 500
    private val _logs = MutableStateFlow<List<LogEntry>>(emptyList())
    val logs: StateFlow<List<LogEntry>> = _logs.asStateFlow()

    fun d(tag: String, message: String) {
        Log.d(tag, message)
        addEntry("D", tag, message)
    }

    fun i(tag: String, message: String) {
        Log.i(tag, message)
        addEntry("I", tag, message)
    }

    fun w(tag: String, message: String, throwable: Throwable? = null) {
        Log.w(tag, message, throwable)
        val msg = if (throwable != null) "$message | ${throwable.message}" else message
        addEntry("W", tag, msg)
    }

    fun e(tag: String, message: String, throwable: Throwable? = null) {
        Log.e(tag, message, throwable)
        val msg = if (throwable != null) "$message | ${throwable::class.simpleName}: ${throwable.message}" else message
        addEntry("E", tag, msg)
    }

    fun clear() {
        _logs.value = emptyList()
    }

    fun export(): String {
        return _logs.value.joinToString("\n") { it.toString() }
    }

    private fun addEntry(level: String, tag: String, message: String) {
        _logs.update { current ->
            val entry = LogEntry(level = level, tag = tag, message = message)
            (listOf(entry) + current).take(MAX_ENTRIES)
        }
    }
}
