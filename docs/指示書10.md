Copyfun downloadFolder(folder: NetworkFile) {
    if (!folder.isDirectory) return
    val server = uiState.value.selectedServer ?: return
    val locationUri = uiState.value.downloadLocationUri
    if (locationUri == null) {
        postStatus("ダウンロード先を設定してください")
        return
    }

    viewModelScope.launch(Dispatchers.IO) {
        postStatus("フォルダをスキャン中: ${folder.name}")
        try {
            val allFiles = collectFilesRecursively(server, folder.path)
            if (allFiles.isEmpty()) {
                postStatus("ダウンロード可能なファイルがありません: ${folder.name}")
                return@launch
            }

            allFiles.forEach { file ->
                // フォルダ構造を維持したファイル名を作成
                val relativePath = file.path.removePrefix(folder.path).removePrefix("/")
                val safeFileName = relativePath.replace("/", "_")

                downloadManager.enqueueDownload(
                    sourceUri = file.path,
                    server = server,
                    destinationUri = locationUri.toUri(),
                    fileName = safeFileName.ifBlank { file.name },
                    totalBytes = file.size
                )
            }

            postStatus("${allFiles.size} 件をキューに追加: ${folder.name}")
        } catch (e: Exception) {
            AppLogger.e("NetworkBrowser", "Folder download scan failed: ${folder.name}", e)
            postStatus("フォルダスキャン失敗: ${e.message}")
        }
    }
}

private suspend fun collectFilesRecursively(
    server: NetworkServer,
    path: String
): List<NetworkFile> {
    val result = mutableListOf<NetworkFile>()
    val queue = ArrayDeque<String>()
    queue.add(path)

    while (queue.isNotEmpty()) {
        val currentPath = queue.removeFirst()
        val files = runCatching { listFiles(server, currentPath) }.getOrElse { emptyList() }
        for (file in files) {
            if (file.isDirectory) {
                queue.add(file.path)
            } else {
                result.add(file)
            }
        }
    }

    return result
}

現在の registerComicFolder はフォルダを1つの書籍として登録するだけで、中のZIPファイルを走査していません。

registerComicFolder を以下に完全置換:

Copyprivate val SUPPORTED_ARCHIVE_EXTENSIONS = setOf("zip", "cbz", "cbr", "rar")

fun registerComicFolder(folderUri: Uri, resolver: ContentResolver) {
    viewModelScope.launch(Dispatchers.IO) {
        val document = DocumentFile.fromTreeUri(context, folderUri)
        if (document == null || !document.canRead()) {
            _uiState.update { it.copy(errorMessage = "このフォルダは読み取れません") }
            return@launch
        }

        runCatching {
            resolver.takePersistableUriPermission(
                folderUri,
                Intent.FLAG_GRANT_READ_URI_PERMISSION
            )
        }

        val children = document.listFiles()

        // フォルダ内のアーカイブファイル (ZIP/CBZ/CBR/RAR) を探す
        val archives = children.filter { child ->
            child.isFile && child.name?.substringAfterLast('.', "")
                ?.lowercase(Locale.getDefault()) in SUPPORTED_ARCHIVE_EXTENSIONS
        }

        // フォルダ内の画像ファイルを探す
        val images = children.filter { child ->
            child.isFile && child.name.hasImageExtension()
        }

        // サブフォルダを探す（各サブフォルダを個別書籍として登録）
        val subFolders = children.filter { child -> child.isDirectory }

        var registeredCount = 0

        // 1) アーカイブファイルをそれぞれ個別に登録
        for (archive in archives) {
            val sourceString = archive.uri.toString()
            if (comicBookDao.findBySourceUri(sourceString) != null) continue

            val title = archive.name ?: "archive"
            val coverUri = try {
                extractFirstImageFromArchive(archive.uri)
            } catch (_: Exception) { null }

            comicBookDao.upsert(
                ComicBook(
                    title = title,
                    sourceUri = sourceString,
                    sourceType = "ARCHIVE",
                    coverUri = coverUri,
                    totalPages = 0
                )
            )
            registeredCount++
            AppLogger.i("ComicReader", "Auto-registered archive: $title")
        }

        // 2) サブフォルダ内の画像があるフォルダをそれぞれ個別に登録
        for (subFolder in subFolders) {
            val hasImages = subFolder.listFiles().any { it.isFile && it.name.hasImageExtension() }
            val hasArchives = subFolder.listFiles().any { child ->
                child.isFile && child.name?.substringAfterLast('.', "")
                    ?.lowercase(Locale.getDefault()) in SUPPORTED_ARCHIVE_EXTENSIONS
            }

            if (hasImages) {
                val sourceString = subFolder.uri.toString()
                if (comicBookDao.findBySourceUri(sourceString) != null) continue

                val title = subFolder.name ?: "folder"
                val coverUri = subFolder.listFiles()
                    .filter { it.isFile && it.name.hasImageExtension() }
                    .minByOrNull { it.name?.lowercase() ?: "" }
                    ?.uri?.toString()

                comicBookDao.upsert(
                    ComicBook(
                        title = title,
                        sourceUri = sourceString,
                        sourceType = "FOLDER",
                        coverUri = coverUri,
                        totalPages = subFolder.listFiles().count { it.isFile && it.name.hasImageExtension() }
                    )
                )
                registeredCount++
                AppLogger.i("ComicReader", "Auto-registered subfolder: $title")
            }

            // サブフォルダ内のアーカイブも登録
            if (hasArchives) {
                for (archive in subFolder.listFiles().filter { child ->
                    child.isFile && child.name?.substringAfterLast('.', "")
                        ?.lowercase(Locale.getDefault()) in SUPPORTED_ARCHIVE_EXTENSIONS
                }) {
                    val sourceString = archive.uri.toString()
                    if (comicBookDao.findBySourceUri(sourceString) != null) continue

                    val title = archive.name ?: "archive"
                    val coverUri = try {
                        extractFirstImageFromArchive(archive.uri)
                    } catch (_: Exception) { null }

                    comicBookDao.upsert(
                        ComicBook(
                            title = title,
                            sourceUri = sourceString,
                            sourceType = "ARCHIVE",
                            coverUri = coverUri,
                            totalPages = 0
                        )
                    )
                    registeredCount++
                    AppLogger.i("ComicReader", "Auto-registered archive in subfolder: $title")
                }
            }
        }

        // 3) ルートに画像しかないフォルダ自体を登録（アーカイブもサブフォルダもなく画像だけの場合）
        if (archives.isEmpty() && subFolders.isEmpty() && images.isNotEmpty()) {
            val sourceString = folderUri.toString()
            if (comicBookDao.findBySourceUri(sourceString) == null) {
                val title = document.name ?: "comic"
                val coverUri = images
                    .minByOrNull { it.name?.lowercase() ?: "" }
                    ?.uri?.toString()

                comicBookDao.upsert(
                    ComicBook(
                        title = title,
                        sourceUri = sourceString,
                        sourceType = "FOLDER",
                        coverUri = coverUri,
                        totalPages = images.size
                    )
                )
                registeredCount++
            }
        }

        if (registeredCount == 0) {
            _uiState.update { it.copy(errorMessage = "登録できるコミックが見つかりませんでした") }
        } else {
            AppLogger.i("ComicReader", "Registered $registeredCount books from folder")
        }
    }
}

修正3: Android 4 対応について
対応不可 です。理由は以下の通りです。

Android 4.x は API レベル 14〜19 です。このアプリが使用している技術スタックはすべて Android 5.0 (API 21) 以上を必須としています。

Jetpack Compose の最低要件は API 21 です。さらに、現在使用中のほぼすべてのライブラリ（AndroidX, Hilt, Room, ExoPlayer/Media3, Coil, DataStore, WorkManager 等）は API 21 以上を要求しています。2023年10月以降、AndroidX ライブラリは API 19 未満のサポートを完全に打ち切っており、仮に API 19 まで下げたとしてもCompose自体が動きません。

UI フレームワーク、DI、データベース、メディア再生、画像ロードの全レイヤーが Compose/AndroidX に依存しているため、機能・UIを変えずに Android 4 対応することは技術的に不可能です。
