        val videoCount: Int
    )

    val deleteTarget: StateFlow<VideoFolder?> = _deleteTarget

    val folders: StateFlow<List<FolderState>> = combine(
        folderDao.getAllFolders(),
        lockConfigDao.getConfigsByType(LockTargetType.VIDEO_FOLDER)
    ) { folders, locks ->
        folders.map { folder ->
            val lockConfig = locks.find { it.targetId == folder.id }
            FolderState(
                folder = folder,
                isLocked = lockConfig?.isEnabled ?: false,
                videoCount = countVideosInFolder(folder)
            )
        }
    }.stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), emptyList())

    fun addFolder(uri: Uri) {
        viewModelScope.launch {
            // 永続パーミッション取得
            context.contentResolver.takePersistableUriPermission(
                uri,
                Intent.FLAG_GRANT_READ_URI_PERMISSION
            )
            val displayName = getDisplayNameFromUri(uri)
            folderDao.insertFolder(
                VideoFolder(
                    uri = uri.toString(),
                    displayName = displayName,
                    addedDate = System.currentTimeMillis()
                )
            )
        }
    }

    fun deleteFolder(folder: VideoFolder) {
        viewModelScope.launch {
            try {
                context.contentResolver.releasePersistableUriPermission(
                    folder.uri.toUri(),
                    Intent.FLAG_GRANT_READ_URI_PERMISSION
                )
            } catch (_: Exception) {}
            lockConfigDao.deleteByTarget(LockTargetType.VIDEO_FOLDER, folder.id)
            folderDao.deleteFolder(folder)
        }
    }

    fun toggleLock(folder: VideoFolder) {
        viewModelScope.launch {
            val existing = lockConfigDao.getConfig(
                LockTargetType.VIDEO_FOLDER, folder.id
            )
            if (existing != null) {
                lockConfigDao.upsertConfig(
                    existing.copy(isEnabled = !existing.isEnabled)
                )
            } else {
                // 初回ロック設定 → PIN設定フローへ誘導
                // UIイベントを発行
            }
        }
    }

## 修正3: サムネイル表示の修正

動画サムネイルが実際に表示されない問題を修正してください
問題
サムネイル枠は表示されるが、中身の画像がロードされていない。

原因
おそらく以下のいずれか:

サムネイル取得処理が非同期で完了前に表示が終わっている
SAF経由のURIに対してMediaMetadataRetrieverが動作していない
Bitmapがnullのまま渡されている
修正方針
Coilを使ってサムネイルを取得・キャッシュ・表示する仕組みに統一する。 独自のサムネイルローダーではなくCoilのVideoFrameDecoder を使用。

SAF URI でのサムネイル取得の注意
SAF(content://) URI の場合、CoilのVideoFrameDecoderが直接 読めない可能性がある。その場合のフォールバック処理:

Copy// カスタム Fetcher を追加
class SafVideoThumbnailFetcher(
    private val context: Context,
    private val uri: Uri
) : Fetcher {
    override suspend fun fetch(): FetchResult {
        // ContentResolver 経由でサムネイル取得
        val bitmap = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
            context.contentResolver.loadThumbnail(uri, Size(320, 180), null)
        } else {
            // ParcelFileDescriptor → MediaMetadataRetriever
            context.contentResolver.openFileDescriptor(uri, "r")?.use { pfd ->
                MediaMetadataRetriever().use { retriever ->
                    retriever.setDataSource(pfd.fileDescriptor)
                    retriever.getFrameAtTime(
                        1_000_000,
                        MediaMetadataRetriever.OPTION_CLOSEST_SYNC
                    )
                }
            }
        }
        return SourceFetchResult(
            source = bitmap?.toImageSource() ?: throw IOException("Thumbnail failed"),
            mimeType = "image/jpeg",
            dataSource = DataSource.DISK
        )
    }

    class Factory(private val context: Context) : Fetcher.Factory<Uri> {
        override fun create(data: Uri, options: Options, imageLoader: ImageLoader): Fetcher? {
            if (data.scheme == "content") {
                return SafVideoThumbnailFetcher(context, data)
            }
            return null
        }
    }
}

    fun enterFullScreen(activity: ComponentActivity) {
        // 1. 横画面固定
        activity.requestedOrientation =
            ActivityInfo.SCREEN_ORIENTATION_SENSOR_LANDSCAPE

        // 2. システムバー非表示
        val window = activity.window
        WindowCompat.setDecorFitsSystemWindows(window, false)
        WindowCompat.getInsetsController(window, window.decorView).apply {
            hide(WindowInsetsCompat.Type.systemBars())
            systemBarsBehavior =
                WindowInsetsControllerCompat.BEHAVIOR_SHOW_TRANSIENT_BARS_BY_SWIPE
        }

        // 3. ノッチ領域にも描画
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.P) {
            window.attributes = window.attributes.apply {
                layoutInDisplayCutoutMode =
                    WindowManager.LayoutParams.LAYOUT_IN_DISPLAY_CUTOUT_MODE_SHORT_EDGES
            }
        }

        // 4. 画面常時ON
        window.addFlags(WindowManager.LayoutParams.FLAG_KEEP_SCREEN_ON)
    }

    fun exitFullScreen(activity: ComponentActivity) {
        val window = activity.window
        WindowCompat.setDecorFitsSystemWindows(window, true)
        WindowCompat.getInsetsController(window, window.decorView).apply {
            show(WindowInsetsCompat.Type.systemBars())
        }
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.P) {
            window.attributes = window.attributes.apply {
                layoutInDisplayCutoutMode =
                    WindowManager.LayoutParams.LAYOUT_IN_DISPLAY_CUTOUT_MODE_DEFAULT
            }
        }
        window.clearFlags(WindowManager.LayoutParams.FLAG_KEEP_SCREEN_ON)
        activity.requestedOrientation =
            ActivityInfo.SCREEN_ORIENTATION_UNSPECIFIED
    }

fun PlayerScreen(
    videoUri: String,
    viewModel: PlayerViewModel = hiltViewModel()
) {
    val context = LocalContext.current
    val activity = context as ComponentActivity

    // 全画面の進入と退出
    DisposableEffect(Unit) {
        FullScreenUtil.enterFullScreen(activity)
        onDispose {
            FullScreenUtil.exitFullScreen(activity)
        }
    }

    // ExoPlayer のセットアップ
    val exoPlayer = remember {
        ExoPlayer.Builder(context)
            .setSeekBackIncrementMs(
                viewModel.seekInterval.value.toLong() * 1000
            )
            .setSeekForwardIncrementMs(
                viewModel.seekInterval.value.toLong() * 1000
            )
            .build()
    }

    DisposableEffect(exoPlayer) {
        val mediaItem = MediaItem.fromUri(videoUri)
        exoPlayer.setMediaItem(mediaItem)
        exoPlayer.prepare()
        exoPlayer.playWhenReady = true
        onDispose {
            exoPlayer.release()
        }
    }

    // 全画面プレイヤーUI
    Box(modifier = Modifier.fillMaxSize().background(Color.Black)) {
        // PlayerView
        AndroidView(
            factory = { ctx ->
                PlayerView(ctx).apply {
                    player = exoPlayer
                    useController = false  // 独自コントロールを使用
                    resizeMode = AspectRatioFrameLayout.RESIZE_MODE_FIT
                    layoutParams = FrameLayout.LayoutParams(
                        ViewGroup.LayoutParams.MATCH_PARENT,
                        ViewGroup.LayoutParams.MATCH_PARENT
                    )
                }
            },
            modifier = Modifier.fillMaxSize()
        )

        // ジェスチャーオーバーレイ

) {

Copy@HiltViewModel
class NetworkBrowserViewModel @Inject constructor(
    private val smbClientProvider: SmbClientProvider,
    private val downloadManager: NetworkDownloadManager,
    @ApplicationContext private val context: Context
) : ViewModel() {

    private val _currentPath = MutableStateFlow("/")
    val currentPath: StateFlow<String> = _currentPath

    private val _items = MutableStateFlow<List<NetworkFile>>(emptyList())
    val items: StateFlow<List<NetworkFile>> = _items

    private val _loading = MutableStateFlow(false)
    val loading: StateFlow<Boolean> = _loading

    private val _error = MutableStateFlow<String?>(null)
    val error: StateFlow<String?> = _error

    private val _selectedItems = MutableStateFlow<Set<String>>(emptySet())
    private var server: NetworkServer? = null
    private var diskShare: DiskShare? = null

    fun connect(server: NetworkServer) {
        this.server = server
        navigateTo("/")
    }

    fun navigateTo(path: String) {
        viewModelScope.launch {
            _loading.value = true
            _error.value = null
            _currentPath.value = path
            try {
                _items.value = withContext(Dispatchers.IO) {
                    listFiles(path)
                }.sortedWith(
                    compareByDescending<NetworkFile> { it.isDirectory }
                        .thenBy { it.name.lowercase() }
                )
            } catch (e: Exception) {
                _error.value = "読み込み失敗: ${e.localizedMessage}"
            } finally {
                _loading.value = false
            }
        }
    }

    private suspend fun listFiles(path: String): List<NetworkFile> {
        val srv = server ?: throw IllegalStateException("Not connected")
        return when (srv.protocol) {
            Protocol.SMB -> listSmb(srv, path)
            Protocol.HTTP, Protocol.WEBDAV -> listHttp(srv, path)
        }
    }

    private fun listSmb(srv: NetworkServer, path: String): List<NetworkFile> {
        val share = getOrCreateShare(srv)
        return share.list(path.removePrefix("/")).map { info ->
            NetworkFile(
                name = info.fileName,
                path = "$path/${info.fileName}".replace("//", "/"),
                size = info.endOfFile,
                isDirectory = EnumWithValue.EnumUtils
                    .isSet(info.fileAttributes, FileAttributes.FILE_ATTRIBUTE_DIRECTORY),
                lastModified = info.changeTime.toEpochMillis()
            )
        }.filter { it.name != "." && it.name != ".." }
    }

    fun canGoUp(): Boolean = _currentPath.value != "/"

    fun navigateUp() {
        val parent = _currentPath.value
            .trimEnd('/')
            .substringBeforeLast('/', "/")
        navigateTo(parent.ifEmpty { "/" })
    }

    fun downloadFile(file: NetworkFile) {
        val srv = server ?: return
        viewModelScope.launch {
            downloadManager.enqueue(
                server = srv,
                remotePath = file.path,
                fileName = file.name,
                fileSize = file.size
            )
        }
    }

    fun getStreamUri(file: NetworkFile): StreamableUri {
        val srv = server ?: throw IllegalStateException()
        return StreamableUri(
            server = srv,
            remotePath = file.path,
            fileName = file.name
        )
    }
}

Copyobject LocaleHelper {

    enum class AppLanguage(val code: String, val displayName: String) {
        JAPANESE("ja", "日本語"),
        ENGLISH("en", "English"),
        SYSTEM("", "システムに従う / Follow System")
    }

    fun applyLanguage(context: Context, language: AppLanguage) {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
            // Android 13+: Per-App Language API
            val localeManager = context.getSystemService(LocaleManager::class.java)
            if (language == AppLanguage.SYSTEM) {
                localeManager.applicationLocales = LocaleList.getEmptyLocaleList()
            } else {
                localeManager.applicationLocales =
                    LocaleList.forLanguageTags(language.code)
            }
        } else {
            // Android 12以下: AppCompatDelegate
            val tag = if (language == AppLanguage.SYSTEM) "" else language.code
            AppCompatDelegate.setApplicationLocales(
                LocaleListCompat.forLanguageTags(tag)
            )
        }
    }

    fun getCurrentLanguage(context: Context): AppLanguage {
        val locales = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
            context.getSystemService(LocaleManager::class.java).applicationLocales
        } else {
            AppCompatDelegate.getApplicationLocales()
                .let { LocaleList.forLanguageTags(it.toLanguageTags()) }
        }

        if (locales.isEmpty) return AppLanguage.SYSTEM

        return when (locales[0]?.language) {
            "ja" -> AppLanguage.JAPANESE
            "en" -> AppLanguage.ENGLISH
            else -> AppLanguage.SYSTEM
        }
    }
}
