問題1: ログビューアを開くとクラッシュする
原因特定:

LogViewerSheet.kt の LazyColumn で key に "${entry.timestamp}-${entry.tag}-${entry.level}" を使っている。同一ミリ秒内に同じタグ・同じレベルでログが複数追加されると キーが重複 し、LazyColumn が IllegalArgumentException を投げてクラッシュする。

加えて、AppLogger の init(context) メソッドは存在しない。LiteMediaApplication.kt の AppLogger.init(this) はコンパイル時にエラーにはなっていない（init が何かの形で定義されているか、あるいはファイルログ機能は削除されたが呼び出しだけ残っている可能性がある）。現在の AppLogger.kt にはファイル書き込みもない。つまりクラッシュしたらログは全部消える。

import android.content.Intent

object AppLogger {
    data class LogEntry(
        val id: Long,
        val timestamp: Long = System.currentTimeMillis(),
        val level: String,
        val tag: String,
        val message: String
    ) {
        val timeString: String
            get() = SimpleDateFormat("HH:mm:ss.SSS", Locale.getDefault()).format(Date(timestamp))

        val dateTimeString: String
            get() = SimpleDateFormat("yyyy-MM-dd HH:mm:ss.SSS", Locale.getDefault()).format(Date(timestamp))

        override fun toString(): String = "[$dateTimeString] $level/$tag: $message"
    }

    private const val MAX_MEMORY_ENTRIES = 500
    private const val MAX_FILE_SIZE_BYTES = 2 * 1024 * 1024L // 2MB
    private const val LOG_FILE_NAME = "litemedia_log.txt"
    private const val LOG_BACKUP_NAME = "litemedia_log_prev.txt"

    private val idCounter = AtomicLong(0)
    private val _logs = MutableStateFlow<List<LogEntry>>(emptyList())
    val logs: StateFlow<List<LogEntry>> = _logs.asStateFlow()

    private var logFile: File? = null
    private var logDir: File? = null
    private var appContext: Context? = null
    private val originalHandler = Thread.getDefaultUncaughtExceptionHandler()

    fun init(context: Context) {
        appContext = context.applicationContext
        logDir = File(context.filesDir, "logs").also { it.mkdirs() }
        logFile = File(logDir, LOG_FILE_NAME)

        // クラッシュ時にスタックトレースをファイルに書き込む
        Thread.setDefaultUncaughtExceptionHandler { thread, throwable ->
            try {
                val sw = java.io.StringWriter()
                throwable.printStackTrace(PrintWriter(sw))
                val crashMsg = "UNCAUGHT EXCEPTION in ${thread.name}: ${sw}"
                writeToFile("CRASH", "UncaughtException", crashMsg)
            } catch (_: Exception) { }
            originalHandler?.uncaughtException(thread, throwable)
        }

        // 起動ログ
        i("AppLogger", "Logger initialized. logFile=${logFile?.absolutePath}")
    }

    fun d(tag: String, message: String) {
        Log.d(tag, message)
        addEntry("D", tag, message)
    }

    fun i(tag: String, message: String) {
        Log.i(tag, message)
        addEntry("I", tag, message)
    }

    fun w(tag: String, message: String, throwable: Throwable? = null) {
        if (throwable == null) Log.w(tag, message) else Log.w(tag, message, throwable)
        val msg = if (throwable != null) "$message | ${throwable::class.simpleName}: ${throwable.message}" else message
        addEntry("W", tag, msg)
    }

    fun e(tag: String, message: String, throwable: Throwable? = null) {
        if (throwable == null) Log.e(tag, message) else Log.e(tag, message, throwable)
        val msg = if (throwable != null) {
            val sw = java.io.StringWriter()
            throwable.printStackTrace(PrintWriter(sw))
            "$message | $sw"
        } else message
        addEntry("E", tag, msg)
    }

    fun clear() {
        _logs.value = emptyList()
    }

    fun export(): String {
        // ファイルがあればファイルから返す（クラッシュログも含む）
        val file = logFile
        if (file != null && file.exists() && file.length() > 0) {
            return try { file.readText() } catch (_: Exception) { "" }
        }
        return _logs.value.joinToString("\n") { it.toString() }
    }

    /** Google ドライブ等に共有する Intent を返す */
    fun createShareIntent(context: Context): Intent? {
        val file = logFile ?: return null
        if (!file.exists() || file.length() == 0L) return null

        val authority = "${context.packageName}.logprovider"
        val contentUri = FileProvider.getUriForFile(context, authority, file)

        return Intent(Intent.ACTION_SEND).apply {
            type = "text/plain"
            putExtra(Intent.EXTRA_STREAM, contentUri)
            putExtra(Intent.EXTRA_SUBJECT, "LiteMedia Player - Debug Log")
            addFlags(Intent.FLAG_GRANT_READ_URI_PERMISSION)
        }
    }

    /** バックアップも含めた共有 (複数ファイル) */
    fun getLogFiles(): List<File> {
        val dir = logDir ?: return emptyList()
        return dir.listFiles()?.filter { it.extension == "txt" }?.sortedByDescending { it.lastModified() } ?: emptyList()
    }

    private fun addEntry(level: String, tag: String, message: String) {
        val entry = LogEntry(
            id = idCounter.incrementAndGet(),
            level = level,
            tag = tag,
            message = message
        )
        _logs.update { current ->
            (listOf(entry) + current).take(MAX_MEMORY_ENTRIES)
        }
        writeToFile(level, tag, message)
    }

    private fun writeToFile(level: String, tag: String, message: String) {
        val file = logFile ?: return
        try {
            // ローテーション
            if (file.exists() && file.length() > MAX_FILE_SIZE_BYTES) {
                val backup = File(logDir, LOG_BACKUP_NAME)
                if (backup.exists()) backup.delete()
                file.renameTo(backup)
                logFile = File(logDir, LOG_FILE_NAME)
            }
            val ts = SimpleDateFormat("yyyy-MM-dd HH:mm:ss.SSS", Locale.getDefault()).format(Date())
            FileWriter(logFile, true).use { writer ->
                writer.appendLine("[$ts] $level/$tag: $message")
            }
        } catch (_: Exception) { }
    }
}

問題2: プレイヤータブが異常に遅い（フォルダ切替で10秒待つ）
根本原因の特定（コードを読んだ結果）:

3つの原因が複合している。

原因A: PlayerScreen の FilterChip の onClick で viewModel.selectFolder(folder.id) と viewModel.syncFolder(folder) を両方呼んでいる。syncFolder はキャッシュを削除してから再スキャンする。つまり毎回キャッシュを捨てて DocumentFile.listFiles() をやり直している。SAF の listFiles() はファイル数が多いと非常に遅い（数百ファイルで数秒かかる）。

原因B: resolveVideoItems() の中で BFS で DocumentFile.listFiles() を再帰呼び出ししている。DocumentFile.listFiles() は内部で ContentResolver.query() を毎回呼ぶため、ファイル1つにつき1クエリが走る。100ファイルあると100回クエリ。

原因C: loadMetadataForFolder() が metadataLoading を AtomicBoolean で排他しているが、フォルダAのメタデータ取得中にフォルダBに切り替えると、フォルダBのメタデータ取得が完全にスキップされる。

修正方針:

syncFolder でのキャッシュ強制削除をやめる。通常の切り替えでは selectFolder だけ呼ぶ。
syncFolder はユーザが明示的にリフレッシュしたい場合のみ（長押しか別ボタン）。
resolveVideoItems() を ContentResolver.query() 直接に置き換え、1クエリでフォルダ内の全ファイルを取得。
メタデータのロックをフォルダ単位にする。

Copyprivate fun resolveVideoItems(treeUriString: String): List<VideoItem> {
    videoItemsCache[treeUriString]?.let { return it }

    AppLogger.d("PlayerVM", "Scanning folder: $treeUriString")
    val scanStart = System.currentTimeMillis()

    val treeUri = Uri.parse(treeUriString)
    val childrenUri = DocumentsContract.buildChildDocumentsUriUsingTree(
        treeUri,
        DocumentsContract.getTreeDocumentId(treeUri)
    )

    val files = mutableListOf<VideoItem>()
    val projection = arrayOf(
        DocumentsContract.Document.COLUMN_DOCUMENT_ID,
        DocumentsContract.Document.COLUMN_DISPLAY_NAME,
        DocumentsContract.Document.COLUMN_SIZE,
        DocumentsContract.Document.COLUMN_LAST_MODIFIED,
        DocumentsContract.Document.COLUMN_MIME_TYPE
    )

    // 再帰的に探索する (BFS)
    val queue = ArrayDeque<Uri>()
    queue.add(childrenUri)

    while (queue.isNotEmpty()) {
        val currentUri = queue.removeFirst()
        context.contentResolver.query(currentUri, projection, null, null, null)?.use { cursor ->
            val idCol = cursor.getColumnIndexOrThrow(DocumentsContract.Document.COLUMN_DOCUMENT_ID)
            val nameCol = cursor.getColumnIndexOrThrow(DocumentsContract.Document.COLUMN_DISPLAY_NAME)
            val sizeCol = cursor.getColumnIndexOrThrow(DocumentsContract.Document.COLUMN_SIZE)
            val modifiedCol = cursor.getColumnIndexOrThrow(DocumentsContract.Document.COLUMN_LAST_MODIFIED)
            val mimeCol = cursor.getColumnIndexOrThrow(DocumentsContract.Document.COLUMN_MIME_TYPE)

            while (cursor.moveToNext()) {
                val docId = cursor.getString(idCol)
                val name = cursor.getString(nameCol) ?: continue
                val mime = cursor.getString(mimeCol) ?: ""
                val size = cursor.getLong(sizeCol)
                val modified = cursor.getLong(modifiedCol)

                if (mime == DocumentsContract.Document.MIME_TYPE_DIR) {
                    // サブフォルダ
                    val subUri = DocumentsContract.buildChildDocumentsUriUsingTree(treeUri, docId)
                    queue.add(subUri)
                } else {
                    val ext = name.substringAfterLast('.', "").lowercase()
                    if (ext in SUPPORTED_VIDEO_EXTENSIONS) {
                        val docUri = DocumentsContract.buildDocumentUriUsingTree(treeUri, docId)
                        files.add(
                            VideoItem(
                                uri = docUri,
                                displayName = name,
                                size = size.coerceAtLeast(0L),
                                durationMs = 0L,
                                resolution = null,
                                dateModified = modified.coerceAtLeast(0L)
                            )
                        )
                    }
                }
            }
        }
    }

    val sorted = files.sortedBy { it.displayName.lowercase() }
    val elapsed = System.currentTimeMillis() - scanStart
    AppLogger.d("PlayerVM", "Folder scan complete: ${sorted.size} files (${elapsed}ms)")
    videoItemsCache[treeUriString] = sorted
    return sorted
}

問題3: ZIP コミック読み込みでクラッシュ
根本原因の特定（コードを読んだ結果）:

extractZipImages() 自体は正しく実装されている。問題は その後 の buildPageModelsFromFiles() → processAndMaterializePages() にある。

processAndMaterializePages() が全ページに対して以下を行う:

openStream()?.use { it.readBytes() } — 画像ファイル全体をメモリに読み込む (例: 5MB × 50ページ = 250MB)
BitmapFactory.decodeByteArray() で inJustDecodeBounds チェック
autoTrim も autoSplit も無効でも writeSingleTempPage() で再度ファイルに書き出す
つまり ZIP から一旦 comic_archive_pages に展開し、それをまた全部メモリに読み込んで comic_processed_pages にコピーしている。ページ数が多い大きな ZIP だと、この二重コピーでメモリが枯渇して OOM クラッシュする。

さらに、processAndMaterializePages は withContext(Dispatchers.IO) 内で 1ページずつ逐次処理しているが、GCが間に合わないペースでバイト配列を作成している。

修正方針:

autoTrim/autoSplit が両方無効の場合は、展開済みファイルをそのまま使う（comic_archive_pages のファイル URI をページモデルに直接渡す）。コピーしない。有効な場合のみ processAndMaterializePages を呼ぶが、1ページ処理後に sourceBytes を明示的に null にして GC を促す。

processAndMaterializePages のメモリ対策を強化:

Copyprivate suspend fun processAndMaterializePages(
    sourceName: String,
    openStream: () -> java.io.InputStream?,
    settings: ComicReaderSettings
): List<Uri> {
    return withContext(Dispatchers.IO) {
        try {
            var sourceBytes: ByteArray? = openStream()?.use { it.readBytes() } ?: return@withContext emptyList()
            AppLogger.d("ComicReader", "Processing page: $sourceName (${sourceBytes!!.size} bytes)")

            val bounds = BitmapFactory.Options().apply { inJustDecodeBounds = true }
            BitmapFactory.decodeByteArray(sourceBytes, 0, sourceBytes!!.size, bounds)
            if (bounds.outWidth <= 0 || bounds.outHeight <= 0) {
                AppLogger.w("ComicReader", "Invalid image: $sourceName (${bounds.outWidth}x${bounds.outHeight})")
                sourceBytes = null
                return@withContext emptyList()
            }

            if (!settings.autoTrimEnabled && !settings.autoSplitEnabled) {
                val result = writeSingleTempPage(sourceBytes!!, sourceName)
                sourceBytes = null
                return@withContext result
            }

            val processed = runCatching {
                pageProcessor.process(bytes = sourceBytes!!, settings = settings)
            }.getOrElse { error ->
                AppLogger.w("ComicReader", "PageProcessor failed for $sourceName", error)
                emptyList()
            }

            if (processed.isEmpty()) {
                val result = writeSingleTempPage(sourceBytes!!, sourceName)
                sourceBytes = null
                return@withContext result
            }

            val fullRect = android.graphics.Rect(0, 0, bounds.outWidth, bounds.outHeight)
            if (processed.size == 1 && processed.first().rect == fullRect) {
                val result = writeSingleTempPage(sourceBytes!!, sourceName)
                sourceBytes = null
                return@withContext result
            }

            val maxDim = maxOf(bounds.outWidth, bounds.outHeight)
            val sampleSize = when {
                maxDim > 8_000 -> 4
                maxDim > 4_000 -> 2
                else -> 1
            }
            val decodeOptions = BitmapFactory.Options().apply { inSampleSize = sampleSize }
            val sourceBitmap = BitmapFactory.decodeByteArray(sourceBytes, 0, sourceBytes!!.size, decodeOptions)
            sourceBytes = null  // ← ここで即座に解放

            if (sourceBitmap == null) return@withContext emptyList()

            val scaleX = sourceBitmap.width.toFloat() / bounds.outWidth.toFloat().coerceAtLeast(1f)
            val scaleY = sourceBitmap.height.toFloat() / bounds.outHeight.toFloat().coerceAtLeast(1f)
            val outputUris = mutableListOf<Uri>()

            processed.forEachIndexed { index, page ->
                val rect = page.rect
                val left = (rect.left * scaleX).toInt().coerceIn(0, sourceBitmap.width - 1)
                val top = (rect.top * scaleY).toInt().coerceIn(0, sourceBitmap.height - 1)
                val cropWidth = (rect.width() * scaleX).toInt().coerceIn(1, sourceBitmap.width - left)
                val cropHeight = (rect.height() * scaleY).toInt().coerceIn(1, sourceBitmap.height - top)

                val cropped = Bitmap.createBitmap(sourceBitmap, left, top, cropWidth, cropHeight)
                val file = File(ensureProcessedCacheDir(), "${sourceName.hashCode()}_${System.nanoTime()}_${index}.jpg")
                FileOutputStream(file).use { output ->
                    cropped.compress(Bitmap.CompressFormat.JPEG, 85, output)
                }
                cropped.recycle()
                outputUris += file.toUri()
            }

            sourceBitmap.recycle()
            outputUris
        } catch (_: OutOfMemoryError) {
            AppLogger.e("ComicReader", "OOM processing page: $sourceName")
            System.gc()
            emptyList()
        } catch (error: Exception) {
            AppLogger.e("ComicReader", "Error processing page: $sourceName", error)
            emptyList()
        }
    }
}

    val root = DocumentFile.fromTreeUri(context, folderUri) ?: return emptyList()
    val imageFiles = root.listFiles()
        .filter { file -> file.isFile && file.name.hasImageExtension() }
        .sortedBy { file -> file.name?.lowercase(Locale.getDefault()) }

    val needsProcessing = settings.autoTrimEnabled || settings.autoSplitEnabled
    if (!needsProcessing) {
        return imageFiles.mapIndexed { index, file ->
            ComicPage(
                index = index,
                model = file.uri,
                sourceName = file.name ?: "page"
            )
        }
    }

    return buildPageModelsFromDocuments(imageFiles, settings)
}
